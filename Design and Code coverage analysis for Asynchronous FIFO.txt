module async_fifo_tb;

  parameter FIFO_DEPTH = 16, POINTER = 4;

  reg wclk, rclk, reset;
  reg w_enable, r_enable;
  reg [7:0] wdata;
  reg [7:0] data_in;
  wire [7:0] rdata;
  wire full_flag, empty_flag;

  reg [7:0] fifo [0:FIFO_DEPTH-1];
  reg [POINTER:0] wpointer, rpointer;
  reg [POINTER:0] wpointer_sync, rpointer_sync;
  reg [POINTER:0] Qr, Qw;

  integer count;

  assign full_flag =
    ({~wpointer[POINTER], wpointer[POINTER-1:0]} == rpointer_sync);

  assign empty_flag =
    (wpointer_sync == rpointer);

  always @(posedge wclk or posedge reset) begin
    if (reset)
      wpointer <= 0;
    else if (w_enable && !full_flag) begin
      fifo[wpointer[POINTER-1:0]] <= wdata;
      wpointer <= wpointer + 1;
    end
  end

  always @(posedge rclk or posedge reset) begin
    if (reset)
      rpointer <= 0;
    else if (r_enable && !empty_flag) begin
      rdata <= fifo[rpointer[POINTER-1:0]];
      rpointer <= rpointer + 1;
    end
  end

  always @(posedge rclk or posedge reset) begin
    if (reset) begin
      Qw <= 0;
      wpointer_sync <= 0;
    end else begin
      Qw <= wpointer;
      wpointer_sync <= Qw;
    end
  end

  always @(posedge wclk or posedge reset) begin
    if (reset) begin
      Qr <= 0;
      rpointer_sync <= 0;
    end else begin
      Qr <= rpointer;
      rpointer_sync <= Qr;
    end
  end

  task write_task(input [7:0] d);
    begin
      @(posedge wclk);
      if (!full_flag) begin
        w_enable = 1;
        wdata = d;
        @(posedge wclk);
        w_enable = 0;
      end
    end
  endtask

  task read_task;
    begin
      @(posedge rclk);
      if (!empty_flag) begin
        r_enable = 1;
        @(posedge rclk);
        r_enable = 0;
      end
    end
  endtask

  always #5 wclk = ~wclk;
  always #7 rclk = ~rclk;

  initial begin
    wclk = 0;
    rclk = 0;
    reset = 1;
    w_enable = 0;
    r_enable = 0;

    #15 reset = 0;

    for (count = 0; count < FIFO_DEPTH; count++) begin
      data_in = $random;
      write_task(data_in);
    end

    for (count = 0; count < FIFO_DEPTH; count++) begin
      read_task();
    end

    #100 $finish;
  end

endmodule
